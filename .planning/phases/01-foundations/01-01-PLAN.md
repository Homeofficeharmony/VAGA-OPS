// We replicate the djb2 hash here to verify the contract without needing React.
function dailyIndex(dateStr, poolSize) {
  let hash = 5381;
  for (let i = 0; i < dateStr.length; i++) {
    hash = ((hash << 5) + hash) ^ dateStr.charCodeAt(i);
  }
  hash = hash >>> 0;
  return hash % poolSize;
}
